  <input type="number" id="monsterLevel">

  <label>角色等級</label>
  <input type="number" id="playerLevel">

  <button onclick="calculate()">計算命中率</button>

  <div id="result"></div>

  <script>
    function calculate() {
      const intelligence = parseFloat(document.getElementById("intelligence").value);
      const luck = parseFloat(document.getElementById("luck").value);
      const evasion = parseFloat(document.getElementById("evasion").value);
      const monsterLevel = parseInt(document.getElementById("monsterLevel").value);
      const playerLevel = parseInt(document.getElementById("playerLevel").value);

      if (isNaN(intelligence) || isNaN(luck) || isNaN(evasion) || isNaN(monsterLevel) || isNaN(playerLevel)) {
        document.getElementById("result").innerText = "請輸入所有數值。";
        return;
      }

      const numerator = (intelligence + luck) / 10;
      const levelDiff = monsterLevel - playerLevel;
      const denominator = (evasion + 1) * (1 + 0.04 * levelDiff);
      const hitRate = (numerator / denominator) * 100;

      document.getElementById("result").innerText = `命中率：${hitRate.toFixed(2)}%`;
    }
  </script>
